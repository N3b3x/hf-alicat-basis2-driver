117 DriverResult<InstantaneousData> ReadInstantaneous(uint8_t flow_decimals = 3) noexcept {
118 std::array<uint16_t, 10> regs{};
119 const auto rc = ReadHoldingRegisters(reg::SelectedGas, regs.data(),
120 static_cast<uint8_t>(regs.size()));
121 if (rc != DriverError::None) return DriverResult<InstantaneousData>::failure(rc);
122
123 InstantaneousData out{};
124 out.gas = static_cast<Gas>(regs[0] & 0xFFU);
125
126 const uint16_t status = regs[1];
127 out.mass_flow_overrange = (status & 0x0001U) != 0;
128 out.temperature_overrange = (status & 0x0002U) != 0;
129 out.totalizer_overrange = (status & 0x0004U) != 0;
130 out.valve_held = (status & 0x0008U) != 0;
131 out.valve_thermal_management = (status & 0x0010U) != 0;
132
133 out.raw_temp_2102 = static_cast<int16_t>(regs[2]);
134 out.temperature_c = static_cast<float>(out.raw_temp_2102) * 0.01f;
135
136 out.raw_flow_2103 = static_cast<int16_t>(regs[3]);
137 out.flow = ScaleFlow(out.raw_flow_2103, flow_decimals);
138
139 out.raw_total_volume_2104_2105 =
140 (static_cast<uint32_t>(regs[4]) << 16) | regs[5];
141 out.total_volume = ScaleFlow(static_cast<int32_t>(out.raw_total_volume_2104_2105),
142 flow_decimals);
143
144 out.raw_setpoint_2053_2054 = static_cast<int16_t>(regs[6]);
145 out.setpoint = ScaleFlow(out.raw_setpoint_2053_2054, flow_decimals);
146
147 out.valve_drive_pct = static_cast<float>(static_cast<int16_t>(regs[7])) * 0.01f;
148
149 out.raw_batch_remaining_2108_2109 =
150 (static_cast<uint32_t>(regs[8]) << 16) | regs[9];
151 out.batch_remaining = ScaleFlow(static_cast<int32_t>(out.raw_batch_remaining_2108_2109),
152 flow_decimals);
153
154 return DriverResult<InstantaneousData>::success(out);
155 }

171 DriverResult<InstrumentIdentity> ReadIdentity() noexcept {
172 InstrumentIdentity id{};
173
174 // Firmware version
175 std::array<uint16_t, 1> fw{};
176 if (auto rc = ReadHoldingRegisters(reg::FirmwareVersion, fw.data(), 1);
177 rc != DriverError::None) {
178 return DriverResult<InstrumentIdentity>::failure(rc);
179 }
180 id.firmware_version_raw = fw[0];
181 id.fw_major = static_cast<uint8_t>((fw[0] >> 8) & 0xFFU);
182 id.fw_minor = static_cast<uint8_t>((fw[0] >> 4) & 0x0FU);
183 id.fw_patch = static_cast<uint8_t>(fw[0] & 0x0FU);
184
185 // Serial number
186 std::array<uint16_t, reg::SerialNumberCount> sn{};
187 if (auto rc = ReadHoldingRegisters(reg::SerialNumberStart, sn.data(),
188 reg::SerialNumberCount);
189 rc != DriverError::None) {
190 return DriverResult<InstrumentIdentity>::failure(rc);
191 }
192 for (std::size_t i = 0; i < reg::SerialNumberCount; ++i) {
193 id.serial_number[i * 2] = static_cast<char>((sn[i] >> 8) & 0xFFU);
194 id.serial_number[i * 2 + 1] = static_cast<char>(sn[i] & 0xFFU);
195 }
196 id.serial_number[12] = '\0';
197
198 // Address + unit id + flow units
199 std::array<uint16_t, 5> tail{}; // 45..49
200 if (auto rc = ReadHoldingRegisters(reg::ModbusAddress, tail.data(),
201 static_cast<uint8_t>(tail.size()));
202 rc != DriverError::None) {
203 return DriverResult<InstrumentIdentity>::failure(rc);
204 }
205 id.modbus_address = static_cast<uint8_t>(tail[0] & 0xFFU);
206 id.ascii_unit_id = static_cast<char>(tail[1] & 0xFFU);
207 id.full_scale_flow_raw =
208 (static_cast<uint32_t>(tail[2]) << 16) | tail[3];
209 id.flow_units = static_cast<FlowUnits>(tail[4] & 0xFFU);
210 return DriverResult<InstrumentIdentity>::success(id);
211 }

233 const int32_t scaled = static_cast<int32_t>(

242 return WriteMultipleRegisters(reg::SetpointCommand, regs.data(), 2);

515
527 DriverResult<uint8_t> DiscoverPresentAddresses(uint8_t* present_bitmap,
528 std::size_t bitmap_bytes,
529 uint16_t probe_timeout_ms = 30,
530 uint8_t first_addr = 1,
531 uint8_t last_addr = kMaxModbusAddress) noexcept {
532 if (!present_bitmap || bitmap_bytes < 32) {
533 return DriverResult<uint8_t>::failure(DriverError::BufferTooSmall);
534 }
535 std::memset(present_bitmap, 0, bitmap_bytes);
536
537 const uint8_t saved_addr = address_;
538 const uint16_t saved_timeout = timeout_ms_;
539 timeout_ms_ = probe_timeout_ms;
540
541 uint8_t found = 0;
542 for (uint16_t a = first_addr; a <= last_addr; ++a) {
543 address_ = static_cast<uint8_t>(a);
544 std::array<uint16_t, 1> dummy{};
545 if (ReadHoldingRegisters(reg::FirmwareVersion, dummy.data(), 1) ==
546 DriverError::None) {
547 present_bitmap[a / 8] |= static_cast<uint8_t>(1U << (a % 8));
548 ++found;
549 }
550 }
551
552 address_ = saved_addr;
553 timeout_ms_ = saved_timeout;
554 return DriverResult<uint8_t>::success(found);
555 }

560
561 DriverError ReadHoldingRegisters(uint16_t reg_addr, uint16_t* out, uint8_t count) noexcept {
562 if (!out || count == 0 || count > 125) return DriverError::InvalidParameter;
563 std::array<uint8_t, 8> tx{};
564 const std::size_t tx_len = modbus::BuildReadHolding(address_, reg_addr, count,
565 tx.data(), tx.size());
566 if (tx_len == 0) return DriverError::BufferTooSmall;
567
568 uart_.flush_rx();
569 uart_.write(tx.data(), tx_len);
570
571 // Expected response: addr(1) + fn(1) + bytecount(1) + payload(2*count) + crc(2)
572 const std::size_t need = 5 + 2U * count;
573 std::array<uint8_t, 5 + 2 * 32> rx{}; // up to 32 regs per call
574 if (need > rx.size()) return DriverError::BufferTooSmall;
575 const std::size_t got = uart_.read(rx.data(), need, timeout_ms_);
576 if (got == 0) return DriverError::Timeout;
577 if (got < 5) return DriverError::InvalidFrame;
578 return modbus::ParseReadHolding(rx.data(), got, address_, count, out);
579 }
580
581 DriverResult<void> WriteSingleRegister(uint16_t reg_addr, uint16_t value) noexcept {
582 std::array<uint8_t, 8> tx{};
583 const std::size_t tx_len = modbus::BuildWriteSingle(address_, reg_addr, value,
584 tx.data(), tx.size());
585 if (tx_len == 0) return DriverResult<void>::failure(DriverError::BufferTooSmall);
586
587 uart_.flush_rx();
588 uart_.write(tx.data(), tx_len);
589
590 // Broadcast → no response.
591 if (address_ == kBroadcastAddress) return DriverResult<void>::success();
592
593 std::array<uint8_t, 8> rx{};
594 const std::size_t got = uart_.read(rx.data(), 8, timeout_ms_);
595 if (got == 0) return DriverResult<void>::failure(DriverError::Timeout);
596 return DriverResult<void>{ modbus::ParseWriteSingleEcho(rx.data(), got,
597 tx.data(), tx_len) };
598 }
599
600 DriverResult<void> WriteMultipleRegisters(uint16_t reg_addr,
601 const uint16_t* regs, uint8_t count) noexcept {
602 if (!regs || count == 0 || count > 32) {
603 return DriverResult<void>::failure(DriverError::InvalidParameter);
604 }
605 std::array<uint8_t, 9 + 32 * 2> tx{};
606 const std::size_t tx_len = modbus::BuildWriteMultiple(address_, reg_addr, regs, count,
607 tx.data(), tx.size());
608 if (tx_len == 0) return DriverResult<void>::failure(DriverError::BufferTooSmall);
609
610 uart_.flush_rx();
611 uart_.write(tx.data(), tx_len);
612
613 if (address_ == kBroadcastAddress) return DriverResult<void>::success();
614
615 std::array<uint8_t, 8> rx{};
616 const std::size_t got = uart_.read(rx.data(), 8, timeout_ms_);
617 if (got == 0) return DriverResult<void>::failure(DriverError::Timeout);
618 return DriverResult<void>{ modbus::ParseWriteMultipleEcho(rx.data(), got,
619 address_, reg_addr, count) };
620 }
