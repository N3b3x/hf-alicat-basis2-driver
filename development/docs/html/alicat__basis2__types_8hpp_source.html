76template <typename T>
77struct DriverResult {
78 T value{};
79 DriverError error{DriverError::None};
80
81 constexpr bool ok() const noexcept { return error == DriverError::None; }
82 constexpr explicit operator bool() const noexcept { return ok(); }
83
84 static constexpr DriverResult success(T v) noexcept { return {v, DriverError::None}; }
