36inline uint16_t Crc16(const uint8_t* data, std::size_t length) noexcept {
37 uint16_t crc = 0xFFFF;
38 for (std::size_t i = 0; i < length; ++i) {
39 crc ^= static_cast<uint16_t>(data[i]);
40 for (uint8_t b = 0; b < 8; ++b) {
41 if (crc & 0x0001U) {
42 crc = static_cast<uint16_t>((crc >> 1) ^ 0xA001U);
43 } else {
44 crc >>= 1;
45 }
46 }
47 }
48 return crc;
49}

52inline std::size_t AppendCrc(uint8_t* frame, std::size_t length) noexcept {
53 const uint16_t crc = Crc16(frame, length);
54 frame[length] = static_cast<uint8_t>(crc & 0x00FFU);
55 frame[length + 1] = static_cast<uint8_t>((crc >> 8) & 0x00FFU);
56 return length + 2;
57}

60inline bool CheckCrc(const uint8_t* frame, std::size_t length) noexcept {
61 if (length < 4) return false;
62 const uint16_t want = Crc16(frame, length - 2);
63 const uint16_t got = static_cast<uint16_t>(frame[length - 2]) |
64 (static_cast<uint16_t>(frame[length - 1]) << 8);
65 return want == got;
66}

74inline std::size_t BuildReadHolding(uint8_t unit_addr,
75 uint16_t reg_addr,
76 uint16_t reg_count,
77 uint8_t* out, std::size_t out_capacity) noexcept {
78 if (out_capacity < 8) return 0;
79 out[0] = unit_addr;
80 out[1] = reg::fn::ReadHolding;
81 out[2] = static_cast<uint8_t>((reg_addr >> 8) & 0xFFU);

83 out[4] = static_cast<uint8_t>((reg_count >> 8) & 0xFFU);
84 out[5] = static_cast<uint8_t>(reg_count & 0xFFU);
85 return AppendCrc(out, 6);

105inline std::size_t BuildWriteMultiple(uint8_t unit_addr,
106 uint16_t reg_addr,
107 const uint16_t* regs, uint8_t count,
108 uint8_t* out, std::size_t out_capacity) noexcept {
109 const std::size_t needed = 9 + count * 2;
110 if (count == 0 || out_capacity < needed) return 0;
111 out[0] = unit_addr;
112 out[1] = reg::fn::WriteMultiple;
113 out[2] = static_cast<uint8_t>((reg_addr >> 8) & 0xFFU);
114 out[3] = static_cast<uint8_t>(reg_addr & 0xFFU);
115 out[4] = static_cast<uint8_t>((count >> 8) & 0xFFU);
116 out[5] = static_cast<uint8_t>(count & 0xFFU);
117 out[6] = static_cast<uint8_t>(count * 2);
118 for (uint8_t i = 0; i < count; ++i) {
119 out[7 + i * 2] = static_cast<uint8_t>((regs[i] >> 8) & 0xFFU);
120 out[7 + i * 2 + 1] = static_cast<uint8_t>(regs[i] & 0xFFU);
121 }
122 return AppendCrc(out, 7 + count * 2);
123}

131inline DriverError ParseReadHolding(const uint8_t* frame, std::size_t length,
132 uint8_t expected_addr, uint8_t expected_count,
133 uint16_t* regs) noexcept {
134 if (length < 5) return DriverError::InvalidFrame;
135 if (!CheckCrc(frame, length)) return DriverError::BadCrc;
136 if (frame[0] != expected_addr) return DriverError::ResponseMismatch;
137 if (frame[1] & reg::fn::ExceptionMask) return DriverError::ModbusException;
138 if (frame[1] != reg::fn::ReadHolding) return DriverError::ResponseMismatch;
139 const std::size_t byte_count = frame[2];
140 if (byte_count != expected_count * 2U) return DriverError::InvalidFrame;
141 if (length < 3 + byte_count + 2) return DriverError::InvalidFrame;
142 for (uint8_t i = 0; i < expected_count; ++i) {
143 regs[i] = (static_cast<uint16_t>(frame[3 + i * 2]) << 8) |
144 static_cast<uint16_t>(frame[3 + i * 2 + 1]);
145 }
146 return DriverError::None;
147}

151inline DriverError ParseWriteSingleEcho(const uint8_t* frame, std::size_t length,
152 const uint8_t* request, std::size_t req_len) noexcept {
153 if (length < req_len) return DriverError::InvalidFrame;
154 if (!CheckCrc(frame, length)) return DriverError::BadCrc;
155 if (frame[0] != request[0]) return DriverError::ResponseMismatch;
156 if (frame[1] & reg::fn::ExceptionMask) return DriverError::ModbusException;
157 for (std::size_t i = 0; i < req_len; ++i) {
158 if (frame[i] != request[i]) return DriverError::ResponseMismatch;
159 }
160 return DriverError::None;
161}

164inline DriverError ParseWriteMultipleEcho(const uint8_t* frame, std::size_t length,
165 uint8_t expected_addr,
166 uint16_t expected_reg_addr,
167 uint16_t expected_count) noexcept {
168 if (length < 8) return DriverError::InvalidFrame;
169 if (!CheckCrc(frame, length)) return DriverError::BadCrc;
170 if (frame[0] != expected_addr) return DriverError::ResponseMismatch;
171 if (frame[1] & reg::fn::ExceptionMask) return DriverError::ModbusException;
172 if (frame[1] != reg::fn::WriteMultiple) return DriverError::ResponseMismatch;
173 const uint16_t got_addr = (static_cast<uint16_t>(frame[2]) << 8) | frame[3];
174 const uint16_t got_cnt = (static_cast<uint16_t>(frame[4]) << 8) | frame[5];
175 if (got_addr != expected_reg_addr) return DriverError::ResponseMismatch;
176 if (got_cnt != expected_count) return DriverError::ResponseMismatch;
177 return DriverError::None;
178}
